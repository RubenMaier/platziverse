
COMANDOS PARA ACCEDER POR TERMINAL A CREAR COSAS EN LA DB:

psql postgres
CREATE ROLE platzi WITH LOGIN PASSWORD 'platzi';
CREATE DATABASE platziverse;
GRANT ALL PRIVILEGES ON DATABASE platziverse TO platzi;
\quit
